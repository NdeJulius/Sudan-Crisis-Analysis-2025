def clean_tweet(text):
    text = re.sub(r"http\S+|www\S+|https\S+", '', str(text), flags=re.MULTILINE)  # Remove URLs
    text = re.sub(r'@\w+', '', text)  # Remove mentions
    text = re.sub(r'#', '', text)  # Remove hashtag symbol only
    text = text.strip()
    return text

# Named Entity Recognition (NER)
from transformers import AutoTokenizer, AutoModelForTokenClassification
import torch

tokenizer = AutoTokenizer.from_pretrained("dslim/bert-base-NER")
model = AutoModelForTokenClassification.from_pretrained("dslim/bert-base-NER")
model.eval()

def extract_entities(text):
    tokens = tokenizer(text, return_tensors="pt", truncation=True, max_length=128)
    with torch.no_grad():
        outputs = model(**tokens)
    predictions = torch.argmax(outputs.logits, dim=2)
    tokens_list = tokenizer.convert_ids_to_tokens(tokens["input_ids"][0])
    entities = []
    current_entity = []
    current_label = None
    for token, label_idx in zip(tokens_list, predictions[0].tolist()):
        label = model.config.id2label[label_idx]
        if label.startswith("B-"):
            if current_entity:
                entities.append((" ".join(current_entity), current_label))
                current_entity = []
            current_entity.append(token.replace("##", ""))
            current_label = label[2:]
        elif label.startswith("I-") and current_label == label[2:]:
            current_entity.append(token.replace("##", ""))
        else:
            if current_entity:
                entities.append((" ".join(current_entity), current_label))

# Relation Extraction (Simple Heuristic)
import spacy
nlp = spacy.load("en_core_web_sm")

def extract_relations(text, entities):
    doc = nlp(text)
    relations = []
    # Simple heuristic: consider pairs of entities in the same sentence with verbs linking them
    for sent in doc.sents:
        sent_text = sent.text
        sent_entities = [ent for ent in entities if ent[0] in sent_text]
        if len(sent_entities) < 2:
            continue
        # For each pair, check for verbs connecting them
        for i in range(len(sent_entities)):
            for j in range(i+1, len(sent_entities)):
                ent1 = sent_entities[i][0]
                ent2 = sent_entities[j][0]
                if any(tok.pos_ == "VERB" for tok in sent):
                    relations.append((ent1, "related_to", ent2))
    return relations

df['Relations'] = df.apply(lambda row: extract_relations(row['Clean_Text'], row['Entities']), axis=1)
